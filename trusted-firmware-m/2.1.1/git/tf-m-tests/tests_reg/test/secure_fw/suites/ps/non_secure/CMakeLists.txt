    PRIVATE
        $<$<BOOL:${TFM_NS_MANAGE_NSID}>:ns_test_helpers.c>

)

    PRIVATE
        tfm_config
        tfm_test_framework_ns
        tfm_test_suite_its_ns

)

    INTERFACE
        tfm_test_suite_ps_ns
)
